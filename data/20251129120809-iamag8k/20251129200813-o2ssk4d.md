```python
import cv2
import numpy as np
import os
import time

IMG_SIZE = (640, 640)  # (width, height), such as (1280, 736)
OBJ_THRESH = 0.25
NMS_THRESH = 0.45
CLASSES = ("person", "bicycle", "car","motorbike ","aeroplane ","bus ","train","truck ","boat","traffic light",
           "fire hydrant","stop sign ","parking meter","bench","bird","cat","dog ","horse ","sheep","cow","elephant",
           "bear","zebra ","giraffe","backpack","umbrella","handbag","tie","suitcase","frisbee","skis","snowboard","sports ball","kite",
           "baseball bat","baseball glove","skateboard","surfboard","tennis racket","bottle","wine glass","cup","fork","knife ",
           "spoon","bowl","banana","apple","sandwich","orange","broccoli","carrot","hot dog","pizza ","donut","cake","chair","sofa",
           "pottedplant","bed","diningtable","toilet ","tvmonitor","laptop	","mouse	","remote ","keyboard ","cell phone","microwave ",
           "oven ","toaster","sink","refrigerator ","book","clock","vase","scissors ","teddy bear ","hair drier", "toothbrush ")

# !!! 核心修改 !!! 将 RKNN 替换为 RKNNLite
try:
    from rknnlite.api import RKNNLite  # 使用轻量级 API
except ImportError:
    print("错误: 找不到 'rknnlite' 模块。请在 Python 3.8 环境中安装正确的 RKNN Runtime whl 包。")
    # 如果找不到 rknnlite，则使用 None 占位，避免程序崩溃
    RKNNLite = None

# =======================================================================

# =======================================================================

#实际的控制节点

VIDEO_DEVICE = 11

# YOLOv8m 的输入尺寸
MODEL_INPUT_WIDTH = 640
MODEL_INPUT_HEIGHT = 640

# --- 硬件参数配置 (保持不变) ---
print("正在配置摄像头硬件参数...")
os.system(f"v4l2-ctl -d {DEVICE_SUBDEV} --set-ctrl analogue_gain=2000")
os.system(f"v4l2-ctl -d {DEVICE_SUBDEV} --set-ctrl exposure=3000")

if not cap.isOpened():

cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)

# --- 2. RKNN 模型初始化 ---
rknn = None
if RKNNLite:

            rknn = None

        # 初始化运行时环境 (在 NPU 上)
        ret = rknn.init_runtime()
        if ret != 0:
            print("错误: 初始化 NPU 运行时失败！")
            rknn = None
        else:
            print("RKNN 模型加载并初始化成功。")

    except Exception as e:
        print(f"RKNN 初始化异常: {e}")
        rknn = None


# --- 3. 自动白平衡函数 (保持不变) ---
def simple_white_balance(img):
    b, g, r = cv2.split(img)
    b_avg = np.mean(b);
    g_avg = np.mean(g);
    r_avg = np.mean(r)
    k = (b_avg + g_avg + r_avg) / 3
    b_avg = b_avg if b_avg != 0 else 1
    g_avg = g_avg if g_avg != 0 else 1
    r_avg = r_avg if r_avg != 0 else 1
    kb = k / b_avg;
    kg = k / g_avg;
    kr = k / r_avg
    b = np.clip(b * kb, 0, 255).astype(np.uint8)
    g = np.clip(g * kg, 0, 255).astype(np.uint8)
    r = np.clip(r * kr, 0, 255).astype(np.uint8)
    return cv2.merge([b, g, r])

def filter_boxes(boxes, box_confidences, box_class_probs):
    """Filter boxes with object threshold.
    """
    box_confidences = box_confidences.reshape(-1)
    candidate, class_num = box_class_probs.shape

    class_max_score = np.max(box_class_probs, axis=-1)
    classes = np.argmax(box_class_probs, axis=-1)

    _class_pos = np.where(class_max_score* box_confidences >= OBJ_THRESH)
    scores = (class_max_score* box_confidences)[_class_pos]

    boxes = boxes[_class_pos]
    classes = classes[_class_pos]

    return boxes, classes, scores

# --- 4. YOLOv8 预处理函数 (保持不变) ---
def preprocess(img):
    """缩放图像并归一化到模型输入格式 (RGB, 640x640)"""
    resized_img = cv2.resize(img, (MODEL_INPUT_WIDTH, MODEL_INPUT_HEIGHT), interpolation=cv2.INTER_LINEAR)
    img_rgb = cv2.cvtColor(resized_img, cv2.COLOR_BGR2RGB)
    input_data = img_rgb.astype(np.float32)
    input_data = np.expand_dims(input_data, axis=0)  # (1, 640, 640, 3) NHWC format
    return input_data

def nms_boxes(boxes, scores):
    """Suppress non-maximal boxes.
    # Returns
        keep: ndarray, index of effective boxes.
    """
    x = boxes[:, 0]
    y = boxes[:, 1]
    w = boxes[:, 2] - boxes[:, 0]
    h = boxes[:, 3] - boxes[:, 1]

    areas = w * h
    order = scores.argsort()[::-1]

    keep = []
    while order.size > 0:
        i = order[0]
        keep.append(i)

        xx1 = np.maximum(x[i], x[order[1:]])
        yy1 = np.maximum(y[i], y[order[1:]])
        xx2 = np.minimum(x[i] + w[i], x[order[1:]] + w[order[1:]])
        yy2 = np.minimum(y[i] + h[i], y[order[1:]] + h[order[1:]])

        w1 = np.maximum(0.0, xx2 - xx1 + 0.00001)
        h1 = np.maximum(0.0, yy2 - yy1 + 0.00001)
        inter = w1 * h1

        ovr = inter / (areas[i] + areas[order[1:]] - inter)
        inds = np.where(ovr <= NMS_THRESH)[0]
        order = order[inds + 1]
    keep = np.array(keep)
    return keep


def dfl(position):
    """用纯 NumPy 实现 YOLOv8 的 DFL 解码。"""
    x = np.asarray(position, dtype=np.float32)
    n, c, h, w = x.shape
    p_num = 4
    mc = c // p_num
    y = x.reshape(n, p_num, mc, h, w)

    # NumPy Softmax (沿第 2 轴)
    e_x = np.exp(y - np.max(y, axis=2, keepdims=True))
    y = e_x / np.sum(e_x, axis=2, keepdims=True)

    # 累积矩阵和加权求和
    acc_matrix = np.arange(mc, dtype=np.float32).reshape(1, 1, mc, 1, 1)
    y = (y * acc_matrix).sum(axis=2)

    return y

def box_process(position):
    grid_h, grid_w = position.shape[2:4]
    col, row = np.meshgrid(np.arange(0, grid_w), np.arange(0, grid_h))
    col = col.reshape(1, 1, grid_h, grid_w)
    row = row.reshape(1, 1, grid_h, grid_w)
    grid = np.concatenate((col, row), axis=1)
    stride = np.array([IMG_SIZE[1]//grid_h, IMG_SIZE[0]//grid_w]).reshape(1,2,1,1)

    position = dfl(position)
    box_xy  = grid +0.5 -position[:,0:2,:,:]
    box_xy2 = grid +0.5 +position[:,2:4,:,:]
    xyxy = np.concatenate((box_xy*stride, box_xy2*stride), axis=1)

    return xyxy

def post_process(input_data):
    boxes, scores, classes_conf = [], [], []
    defualt_branch=3
    pair_per_branch = len(input_data)//defualt_branch
    # Python 忽略 score_sum 输出
    for i in range(defualt_branch):
        boxes.append(box_process(input_data[pair_per_branch*i]))
        classes_conf.append(input_data[pair_per_branch*i+1])
        scores.append(np.ones_like(input_data[pair_per_branch*i+1][:,:1,:,:], dtype=np.float32))

    def sp_flatten(_in):
        ch = _in.shape[1]
        _in = _in.transpose(0,2,3,1)
        return _in.reshape(-1, ch)

    boxes = [sp_flatten(_v) for _v in boxes]
    classes_conf = [sp_flatten(_v) for _v in classes_conf]
    scores = [sp_flatten(_v) for _v in scores]

    boxes = np.concatenate(boxes)
    classes_conf = np.concatenate(classes_conf)
    scores = np.concatenate(scores)

    # filter according to threshold
    boxes, classes, scores = filter_boxes(boxes, scores, classes_conf)

    # nms
    nboxes, nclasses, nscores = [], [], []
    for c in set(classes):
        inds = np.where(classes == c)
        b = boxes[inds]
        c = classes[inds]
        s = scores[inds]
        keep = nms_boxes(b, s)

        if len(keep) != 0:
            nboxes.append(b[keep])
            nclasses.append(c[keep])
            nscores.append(s[keep])

    if not nclasses and not nscores:
        return None, None, None

    boxes = np.concatenate(nboxes)
    classes = np.concatenate(nclasses)
    scores = np.concatenate(nscores)

    return boxes, classes, scores


# --- 5. YOLOv8 后处理和画框函数 (保持占位符，需自行实现) ---
# **重要提示：您需要根据您的 RKNN 模型输出结构实现这个函数**
def postprocess_and_draw(rknn_outputs, original_frame):
    """
    使用 RKNN 示例脚本的逻辑对 NPU 输出进行后处理和绘图。
    """

    if rknn_outputs and rknn_outputs[0].ndim >= 2:
        print(f"RKNN Output Shape: {rknn_outputs[0].shape}")
    # 假设输出张量是 rknn_outputs[0]
    else:
        print("RKNN Output is invalid or has low dimensions.")

    # 1. 调用 RKNN 示例脚本的 post_process 函数
    # 注意: YOLOv8 模型的 RKNN 输出可能需要特殊处理，这里直接传入 NPU 原始输出
    boxes, classes, scores = post_process(rknn_outputs)

    # 如果没有检测到目标，直接返回原图
    if boxes is None:
        return original_frame

    # 初始化 COCO_test_helper（用于坐标映射）
    # 在这个实时脚本中，我们简化这个工具类，直接进行简单的缩放和映射。

    H, W, _ = original_frame.shape  # 原始视频帧尺寸 (640x480)

    # 2. 坐标映射（从 640x640 映射回 640x480）

    # 假设您的预处理没有进行 letterbox (即非等比缩放或直接 resize)，
    # 且输入图像为 640x640，目标帧为 640x480。

    # 我们必须将 640x640 上的坐标 (boxes) 缩放到原始 640x480 画面。

    # 缩放因子：
    scale_w = W / IMG_SIZE[0]  # 640/640 = 1.0
    scale_h = H / IMG_SIZE[1]  # 480/640 = 0.75 (如果原始帧是 480 高度)

    # YOLOv8 的 boxes 格式通常是 [x1, y1, x2, y2]
    final_boxes = boxes.copy()
    final_boxes[:, [0, 2]] *= scale_w  # 映射 X 坐标
    final_boxes[:, [1, 3]] *= scale_h  # 映射 Y 坐标

    # 3. 绘图
    for box, score, cl in zip(final_boxes, scores, classes):
        # top, left, right, bottom 对应于 x_min, y_min, x_max, y_max
        x1, y1, x2, y2 = [int(_b) for _b in box]

        # 针对人脸模型，类别名称可能只需要 "Face"
        class_name = CLASSES[cl] if cl < len(CLASSES) else "Unknown"

        # 绘制红色矩形框 (BGR: (0, 0, 255))
        cv2.rectangle(original_frame, (x1, y1), (x2, y2), (0, 0, 255), 2)
        cv2.putText(original_frame, f'{class_name}: {score:.2f}', (x1, y1 - 10),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0, 0, 255), 2)

    return original_frame


# --- 6. 主循环 ---
enable_wb = True
enable_flip = True

while True:
    ret, frame = cap.read()
    if not ret:
        break

    display_frame = frame.copy()

    # 1. 软件白平衡处理
    if enable_wb:
        display_frame = simple_white_balance(display_frame)

    # 2. 上下反转画面 (垂直反转)
    if enable_flip:
        display_frame = cv2.flip(display_frame, 0)

    # 3. NPU 推理和人脸识别
    if rknn:
        # 预处理：缩放并转换颜色通道
        # 注意：rknnlite.inference 期望输入是 NHWC 格式 (1, 640, 640, 3)
        input_data = preprocess(display_frame)

        # 推理
        # data_format=['nhwc'] 告诉 rknn 你的输入数据是 NHWC 格式
        outputs = rknn.inference(inputs=[input_data], data_format=['nhwc'])

        # 后处理和绘图
        display_frame = postprocess_and_draw(outputs, display_frame)

    cv2.imshow('frame', display_frame)

    key = cv2.waitKey(1) & 0xFF
    if key == ord('q'):
        break

cap.release()
if rknn:
    rknn.release()  # 释放 NPU 资源
cv2.destroyAllWindows()
```
